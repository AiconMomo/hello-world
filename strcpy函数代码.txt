char *StrCpy(char* sr)
{
                                int length;
		length = 0;
		const char* p1 = src;
		while (*p1++ != 0)
		{
			length++;
		}
		char* dst = (char*)malloc(length + 1);
		for (int i = 0; i < length; i++)
		{
			dst[i] = src[i];
		}
		dst[length] ='\0';
		return dst;//复制字符串函数
}
